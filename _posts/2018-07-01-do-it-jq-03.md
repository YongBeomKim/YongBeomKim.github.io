## Ajax

```javascript
$.ajax({
    url : "js/books.json",
    type : "Post",
    dataType : "json",
    success:function(data){
        var t = "<ul>";
        $.each(data.book, function(i, v){ // i :인덱스, v : value
            t += "<li><img src='img/"+v.path+"alert'"+v.title+"'/></li>";
            // v.path(key값), v.title(key값) 내부 요소를 추출
        t += "</ul>"
        $("#list")
        });
    }
});
```
